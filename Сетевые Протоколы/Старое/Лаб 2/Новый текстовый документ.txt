
1. Функции локального управления

	создать сокет (возвращает дискриптор)
		socket(domain, type, protocol)  
			domain - набор протоколов (TCP/IP - AF_INET)
			type - режим взаимодействия
			protocol - протокол транспортного уровня (TCP - IPPROTO_TCP )
	
	подключения сокета к коммуникационной среде (TCP/IP - связывание с локал. портом) (возвращает 0)
		bind(socketDesc, addr, addrlen)
			socketDesc - дескриптор сокета
			addr - структура содержащая локальный адрес (TCP/IP - структура sockarrd_in)	
			addrlen - размер структуры addr 
	
	структура sockaddr_in 
		sockaddr_in clientService;
		clientService.sin_family = AF_INET;
		clientService.sin_addr.s_addr = inet_addr(addres);
		clientService.sin_port = htons(port);

2. Функции установления связи

	ожидание сервером запросов от клиентов (возвращает 0)
		listen(socketDesc, maxLen)
			socketDesc - дескриптор сокета через который прог. ожидает запросы
			maxLen - макс. длина очереди входящих запросов

	обращение клиента к серверу (возвращает 0)
		connect(socketDesc, addr, arrlen);
			socketDesc - дескриптор сокета
			addr - структура содержащая локальный адрес (TCP/IP - структура sockarrd_in)	
			addrlen - размер структуры addr 

	прием сервером запроса от клиента на установление связи 
		accept(socketDesc, p_addr, p_addrlen);
			socketDesc - дескриптор сокета
			p_addr - структура содержащая адрес сокета клиента
			p_addrlen - размер структуры p_addr 

	функция gethostbyname (возвращает указатель на структуру hostent)
		struct hostent *h = gethostbyname(name);
			name - символическое имя узла сети
				h_name - имя узла.
				h_aliases - список дополнительных имен узла
				h_addrtype - идентификатор протоколов (TCP/IP - AF_INET)
				h_lenght - длина адреса узла. 
				h_addr - область памяти, содержащую адрес узла

3. Функции обмена данными
	передача данных (возвращает количество переданных байт) (если flags = 0, то аналогичен write)
		send(socketDesc, buf, len, flags)
			socketDesc - дескриптор сокета
			buf - область памяти, содержащую передаваемые данные. 
			len - длина передаваемых данных. 
			flags - модифицирует исполнение системного вызова send. 
	
	получение данных (возвращает количество полученных байт) (если flags = 0, то аналогичен read)
		recv(socketDesc, buf, len, flags)
			socketDesc - дескриптор сокета
			buf - область памяти, содержащую принимаемые данные. 
			len - длина принимаемых данных. 
			flags - модифицирует исполнение системного вызова recv

4. Функции закрытия связи

	закрытие сокета
		close(socketDesc )
			socketDesc - дескриптор сокета
	
	экстренное закрытия связи путем "сброса" еще не переданных данных
		shotdown(socketDesc, how)
			socketDesc - дескриптор сокета
			how - действия, выполняемые при очистке системных буферов socket'а:
				0 - сбросить и далее не принимать данные для чтения из socket'а; 
				1 - сбросить и далее не отправлять данные для посылки через socket; 
				2 - сбросить все данные, передаваемые через socket в любом направлении.
